// 基于 MathBox 的“手绘感”数轴初探 var mathbox = mathBox({ plugins: ['core', 'controls', 'cursor', 'mathbox'], controls: { klass: THREE.OrbitControls } }); var three = mathbox.three; three.renderer.setClearColor(new THREE.Color(0x05070a), 1.0); var view = mathbox.cartesian({ range: [[-5, 5], [-1, 1]], scale: [2, 1], }); // 1. 画数轴（带有一点点 3B1B 的优雅青蓝色） view.axis({ axis: 1, width: 4, color: '#58a6ff', }); // 2. 动态刻度 (让刻度随时间微颤，模拟手绘感) view.interval({ width: 11, // 从 -5 到 5 共 11 个点 expr: function (emit, x, i, t) { // 给 Y 轴坐标加一点点随机抖动 var wobble = Math.sin(t * 2 + i) * 0.02; emit(x, wobble); }, channels: 2, }); view.ticks({ width: 3, size: 15, color: '#ffffff', }); // 3. 加上数字标注 (KaTeX 风格) // MathBox 强大的地方在于可以直接把文字挂载到坐标上 // 这里我们先用简单的点代替，后续教你挂载 KaTeX view.point({ color: '#f2cc60', size: 10, });
